# 添加关键词
npm pkg set keywords='["yyc3","ktv","admin-system","next.js","react","typescript","ai","blockchain"]'

# 添加格式化和类型检查脚本
npm pkg set scripts.format="prettier --write \"**/*.{ts,tsx,js,jsx,json,md}\""
